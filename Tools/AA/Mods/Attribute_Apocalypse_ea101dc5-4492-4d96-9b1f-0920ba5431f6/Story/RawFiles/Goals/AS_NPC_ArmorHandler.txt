// If the game starts and the difficulty is Tactician, start giving all appropriate characters their new skills!
IF
RegionStarted(_)
AND
IsHardcoreMode(1)
AND
DB_AS_TacticianSkills(_Char,_Skill)
AND
ObjectExists(_Char,1)
THEN
CharacterAddSkill(_Char,_Skill);

// This needs to be done because some characters don't exist at game start. Hurr.
IF
ObjectEnteredCombat((CHARACTERGUID)_Char,_)
AND
IsHardcoreMode(1)
AND
DB_AS_TacticianSkills(_Char,_Skill)
THEN
CharacterAddSkill(_Char,_Skill);

// If a character is seen, check if they're an NPC and if they need armor stats setup. Then set them.

IF
DB_Sees(_Char1,_Char2)
AND
CharacterIsPlayer(_Char2,0)
AND
IsTagged(_Char2,"ORIGIN",0)	// Origin characters unanimously have their own gear, so don't fake-armour them.
AND
ObjectIsCharacter(_Char2,1)
AND
ObjectGetFlag(_Char2,"AS_NPCArmorSetUp",0)
THEN
SetStoryEvent(_Char2,"AS_NPCSetArmor");
SetStoryEvent(_Char2,"AS_CheckForNullAttributes");
SetVarInteger(_Char2,"AS_Willpower",0);

// A small debug. Should fix Source Vamp not working on Lohse's 'god'.
IF
DB_Sees(_Char1,_Char2)
AND
DB_AS_Gods(_Char2)
THEN
CharacterAddSourcePoints(_Char2,1);


// Do the same as above if the character enters combat
IF
ObjectEnteredCombat((CHARACTERGUID)_Char,_)
AND
CharacterIsPlayer(_Char,0)
AND
ObjectGetFlag(_Char,"AS_NPCArmorSetUp",0)
THEN
SetStoryEvent(_Char,"AS_NPCSetArmor");



IF
CharacterDied(_Char)
AND
CharacterIsPlayer(_Char,0)
THEN
ObjectClearFlag(_Char,"AS_NPCArmorSetUp");

IF
CharacterResurrected(_Char)
AND
CharacterIsPlayer(_Char,0)
THEN
ProcObjectTimer(_Char,"AS_NPCArmorTimer",1500);

PROC
ProcObjectTimerFinished(_Char,"AS_NPCArmorTimer")
THEN
SetStoryEvent(_Char,"AS_NPCSetArmor");


IF
StoryEvent(_Char,"AS_NPCSetArmor")
THEN
RemoveStatus(_Char,"AS_FakeArmorStr");
RemoveStatus(_Char,"AS_FakeArmorFin");
RemoveStatus(_Char,"AS_FakeArmorInt");

// Start with Str
IF
StoryEvent((CHARACTERGUID)_Char,"AS_NPCSetArmor")
AND
ObjectGetFlag(_Char,"AS_NPCArmorSetUp",0)
AND
CharacterGetAttribute(_Char,"Strength",_Str)
AND
CharacterGetAttribute(_Char,"Finesse",_Fin)
AND
CharacterGetAttribute(_Char,"Intelligence",_Int)
AND
_Str > _Int
AND
_Str > _Fin
THEN
ApplyStatus(_Char,"AS_FakeArmorStr",-1.0,1);
ObjectSetFlag(_Char,"AS_NPCArmorSetUp");


// Int
IF
StoryEvent((CHARACTERGUID)_Char,"AS_NPCSetArmor")
AND
ObjectGetFlag(_Char,"AS_NPCArmorSetUp",0)
AND
CharacterGetAttribute(_Char,"Strength",_Str)
AND
CharacterGetAttribute(_Char,"Finesse",_Fin)
AND
CharacterGetAttribute(_Char,"Intelligence",_Int)
AND
_Int > _Str
AND
_Int > _Fin
THEN
ApplyStatus(_Char,"AS_FakeArmorInt",-1.0,1);
ObjectSetFlag(_Char,"AS_NPCArmorSetUp");


// Finish with Fin if the others didn't apply
IF
StoryEvent((CHARACTERGUID)_Char,"AS_NPCSetArmor")
AND
ObjectGetFlag(_Char,"AS_NPCArmorSetUp",0)
THEN
ApplyStatus(_Char,"AS_FakeArmorFin",-1.0,1);
ObjectSetFlag(_Char,"AS_NPCArmorSetUp");



// Remove all of such armour from players
IF
StoryEvent((CHARACTERGUID)_Char,"AS_CheckAttributes")
AND
DB_AS_ArmorStatuses(_Status)
AND
HasActiveStatus(_Char,_Status,1)
AND
CharacterIsPlayer(_Char,1)
THEN
RemoveStatus(_Char,_Status);
EXITSECTION
